import numpy as np import matplotlib.pyplot as plt from utils import * from SceneCoords import * # Render the 3d object from an orthographic viewpoint # Returns a rendered image def orthographic_render_verts(scene, verts, faces, axis='z', zoom=1.0, height=768, width=1024, draw_axes=True): # Numpy format (h, w, c) # Set all pixels to black image = np.zeros((height, width, 3), dtype='float') # Check if pixel is in visible area pixel_visible = lambda x, y: (x > 0) and (x < width) and (y > 0) and (y < height) # Calculate pixels per world units (horizontal and vertical) # and screen coordinates of origin offset_h = width * (1.0 - zoom) / 2.0 offset_v = height * (1.0 - zoom) / 2.0 scene.set_axis(axis) # set that axis scale_h = width / scene.h_len() * zoom scale_v = height / scene.v_len() * zoom origin_h = int(-(scene.h_0) * scale_h + offset_h) origin_v = int(-(scene.v_0) * scale_v + offset_v) # Draw axes if draw_axes: draw_axes_in_image(image, axis, height, width, origin_h, origin_v) # Draw the verts in white for v in verts: image[height-int(v[scene.v_ref]*scale_v + origin_v)-1,int(v[scene.h_ref]*scale_h + origin_h)-1,:] = 1.0 return image
